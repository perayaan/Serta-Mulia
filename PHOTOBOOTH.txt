    <script>
        const video = document.getElementById('pbVideo');
        const countdownEl = document.getElementById('pbCountdown');
        const statusEl = document.getElementById('pbLiveStatus');
        
        let currentThemeColor = '#4A1519';
        let photosArray = [null, null, null, null, null, null];
        
        // Menyimpan array berisi data canvas untuk dirender ulang saat kompilasi video akhir
        let videoFramesBuffer = [[], [], [], [], [], []]; 
        
        let canvasRenderInterval = null;
        let currentIndex = 0;
        let isRetakeActive = false;
        let retakeTargetIdx = null;

        async function initCameraStream() {
            try {
                const stream = await navigator.mediaDevices.getUserMedia({ video: { width: 1280, height: 720 }, audio: false });
                video.srcObject = stream;
            } catch (err) {
                alert("Gagal memuat kamera. Pastikan izin akses aktif.");
            }
        }

        function setThemeStyle(color) {
            currentThemeColor = color;
            document.getElementById('stripLeft').style.backgroundColor = color;
            document.getElementById('stripRight').style.backgroundColor = color;
        }

        function startPhotoboothSequence() {
            document.getElementById('btnActionStart').style.display = 'none';
            document.getElementById('btnActionDownload').style.display = 'none';
            
            photosArray = [null, null, null, null, null, null];
            videoFramesBuffer = [[], [], [], [], [], []];
            
            for(let i=0; i<6; i++) {
                const slot = document.getElementById(`slot_${i}`);
                slot.innerHTML = `Bingkai ${i+1}`;
                slot.classList.remove('retake-active');
            }
            
            currentIndex = 0;
            isRetakeActive = false;
            executeCountdownCycle();
        }

        function executeCountdownCycle() {
            let count = 5;
            countdownEl.innerText = count;
            countdownEl.style.display = 'block';
            statusEl.style.display = 'block';

            const activeIdx = isRetakeActive ? retakeTargetIdx : currentIndex;
            videoFramesBuffer[activeIdx] = [];

            // Rekam frame-by-frame dengan rasio mutlak 3:2 ke memori buffer selama 5 detik countdown
            canvasRenderInterval = setInterval(() => {
                const frameCanvas = document.createElement('canvas');
                frameCanvas.width = 600;
                frameCanvas.height = 400;
                const fCtx = frameCanvas.getContext('2d');
                
                drawAbsoluteRasio32(fCtx, video, 0, 0, 600, 400);
                videoFramesBuffer[activeIdx].push(frameCanvas);
            }, 100); 

            let timer = setInterval(() => {
                count--;
                if (count > 0) {
                    countdownEl.innerText = count;
                } else {
                    clearInterval(timer);
                    clearInterval(canvasRenderInterval);
                    countdownEl.style.display = 'none';
                    statusEl.style.display = 'none';
                    finalizeCapture();
                }
            }, 1000);
        }

        function finalizeCapture() {
            const activeIdx = isRetakeActive ? retakeTargetIdx : currentIndex;

            // Snapshot resolusi tinggi mutlak 2400 x 1600 (3:2)
            const tempCanvas = document.createElement('canvas');
            tempCanvas.width = 2400;
            tempCanvas.height = 1600;
            const tCtx = tempCanvas.getContext('2d');
            
            drawAbsoluteRasio32(tCtx, video, 0, 0, 2400, 1600);
            const dataUri = tempCanvas.toDataURL('image/png');

            photosArray[activeIdx] = dataUri;
            
            // Tampilan di web preview dipastikan tetap FOTO STATIS (diam) biasa
            const slot = document.getElementById(`slot_${activeIdx}`);
            slot.innerHTML = `<img src="${dataUri}" />`;
            slot.classList.remove('retake-active');

            if(!isRetakeActive) {
                currentIndex++;
                if (currentIndex < 6) {
                    executeCountdownCycle();
                } else {
                    toggleControlButtons(true);
                }
            } else {
                toggleControlButtons(true);
            }
        }

        function drawAbsoluteRasio32(ctx, img, x, y, targetW, targetH) {
            let imgWidth = img.videoWidth || 1280;
            let imgHeight = img.videoHeight || 720;
            
            let renderWidth = imgWidth;
            let renderHeight = imgWidth * (targetH / targetW);

            if (renderHeight > imgHeight) {
                renderHeight = imgHeight;
                renderWidth = imgHeight * (targetW / targetH);
            }

            let offsetX = (imgWidth - renderWidth) / 2;
            let offsetY = (imgHeight - renderHeight) / 2;

            ctx.save();
            ctx.translate(x + targetW, y);
            ctx.scale(-1, 1);
            ctx.drawImage(img, offsetX, offsetY, renderWidth, renderHeight, 0, 0, targetW, targetH);
            ctx.restore();
        }

        function triggerRetakeProcess(index) {
            if (photosArray.includes(null) && !isRetakeActive) return;

            for(let i=0; i<6; i++) document.getElementById(`slot_${i}`).classList.remove('retake-active');

            isRetakeActive = true;
            retakeTargetIdx = index;
            document.getElementById(`slot_${index}`).classList.add('retake-active');
            toggleControlButtons(false);

            executeCountdownCycle();
        }

        function toggleControlButtons(show) {
            document.getElementById('btnActionStart').style.display = show ? 'inline-block' : 'none';
            document.getElementById('btnActionDownload').style.display = show ? 'inline-block' : 'none';
            if(show) isRetakeActive = false;
        }

        // FUNGSI ANDALAN: KOMPILASI NATIVE MEDIARECORDER MENCEGAH VIDEO CORRUPT
        async function packageAndDownloadZip() {
            const zip = new JSZip();
            
            // 1. Masukkan 6 Foto Asli Landscape (2400 x 1600) ke ZIP
            for (let i = 0; i < 6; i++) {
                if (photosArray[i]) {
                    const base64Data = photosArray[i].split(',')[1];
                    zip.file(`foto-asli-2400x1600-bingkai-${i+1}.png`, base64Data, {base64: true});
                }
            }

            // 2. Masukkan 1 Cetakan Sepasang Strip Statis HD (1200 x 1800) Rapat Tanpa Celah
            const exportCanvas = document.createElement('canvas');
            exportCanvas.width = 1200;
            exportCanvas.height = 1800;
            const ctxExport = exportCanvas.getContext('2d');

            ctxExport.fillStyle = "#ffffff";
            ctxExport.fillRect(0, 0, 1200, 1800);

            async function drawStripHD(offsetX, startIdx) {
                ctxExport.fillStyle = currentThemeColor;
                ctxExport.fillRect(offsetX, 0, 600, 1800);

                for (let i = 0; i < 3; i++) {
                    const imgData = photosArray[startIdx + i];
                    if (imgData) {
                        const img = await loadImageAsync(imgData);
                        ctxExport.drawImage(img, offsetX + 40, 120 + (i * 480), 520, 346.6);
                    }
                }
                
                ctxExport.fillStyle = "#ffffff";
                ctxExport.font = "italic bold 46px 'Times New Roman', serif";
                ctxExport.textAlign = "center";
                ctxExport.fillText("Elevate Moment's", offsetX + 300, 1680);
            }

            await drawStripHD(0, 0);
            await drawStripHD(600, 3);

            const combinedBase64 = exportCanvas.toDataURL('image/png').split(',')[1];
            zip.file("combined-frame-1200x1800.png", combinedBase64, {base64: true});

            // 3. NATIVE RECORDER SYSTEM: Merajut Video Lembar Cetak Bergerak Serempak
            const videoCanvas = document.createElement('canvas');
            videoCanvas.width = 1200;
            videoCanvas.height = 1800;
            const vCtx = videoCanvas.getContext('2d');

            // Tangkap stream asli dari canvas dengan frame rate 15 FPS konstan
            const videoStream = videoCanvas.captureStream(15);
            const chunks = [];
            
            // Set Bitrate tinggi (5 Mbps) agar kompresi tajam, halus, dan kontainer terenkripsi valid
            const recorder = new MediaRecorder(videoStream, { 
                mimeType: 'video/webm;codecs=vp8',
                videoBitsPerSecond: 5000000 
            });

            recorder.ondataavailable = (e) => { if (e.data.size > 0) chunks.push(e.data); };
            
            const recordPromise = new Promise((resolve) => {
                recorder.onstop = () => {
                    const videoBlob = new Blob(chunks, { type: 'video/webm' });
                    zip.file("live-photobooth-strip-frame.webm", videoBlob);
                    resolve();
                };
            });

            recorder.start();

            // Lakukan render loop sebanyak 50 frame (~5 detik video berjalan stabil)
            for (let f = 0; f < 50; f++) {
                vCtx.fillStyle = "#ffffff";
                vCtx.fillRect(0, 0, 1200, 1800);

                // Sisi Strip Kiri
                vCtx.fillStyle = currentThemeColor;
                vCtx.fillRect(0, 0, 600, 1800);
                for (let i = 0; i < 3; i++) {
                    const frames = videoFramesBuffer[i];
                    if (frames && frames.length > 0) {
                        const sourceCanvas = frames[f % frames.length];
                        vCtx.drawImage(sourceCanvas, 40, 120 + (i * 480), 520, 346.6);
                    }
                }
                vCtx.fillStyle = "#ffffff";
                vCtx.font = "italic bold 46px 'Times New Roman', serif";
                vCtx.textAlign = "center";
                vCtx.fillText("Elevate Moment's", 300, 1680);

                // Sisi Strip Kanan
                vCtx.fillStyle = currentThemeColor;
                vCtx.fillRect(600, 0, 600, 1800);
                for (let i = 0; i < 3; i++) {
                    const frames = videoFramesBuffer[3 + i];
                    if (frames && frames.length > 0) {
                        const sourceCanvas = frames[f % frames.length];
                        vCtx.drawImage(sourceCanvas, 640, 120 + (i * 480), 520, 346.6);
                    }
                }
                vCtx.fillStyle = "#ffffff";
                vCtx.fillText("Elevate Moment's", 900, 1680);

                // Berikan jeda waktu tunggu agar MediaRecorder sempat menangkap buffer frame canvas secara mulus
                await new Promise(r => setTimeout(r, 65));
            }

            recorder.stop();
            await recordPromise;

            // Unduh Berkas ZIP Hasil Akhir
            zip.generateAsync({type:"blob"}).then(function(content) {
                const link = document.createElement('a');
                link.href = URL.createObjectURL(content);
                link.download = "photobooth-absolute-pack.zip";
                link.click();
            });
        }

        function loadImageAsync(src) {
            return new Promise((resolve) => {
                const img = new Image();
                img.onload = () => resolve(img);
                img.src = src;
            });
        }

        window.onload = initCameraStream;
    </script>
