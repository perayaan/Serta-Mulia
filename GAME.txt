            ctx.save();

            ctx.strokeStyle = "#000000";

        // --- LOGIKA UTAMA SIKLUS PERMAINAN (FRAME LOOP) ---
        function loop() {
            if (!isPopupActive && !isPuzzleActive && !isPhotoActive && !isClawPopupActive) {
                if (keys.left) { 
                    player.x -= player.speed; 
                    player.dir = 'left'; 
                } else if (keys.right) { 
                    player.x += player.speed; 
                    player.dir = 'right'; 
                }
                if (keys.up && !player.jumping) { 
                    player.vy = -15; 
                    player.jumping = true; 
                }
            }
            
            player.vy += 0.8; 
            player.y += player.vy;
            if (player.y > 0) { player.y = 0; player.vy = 0; player.jumping = false; }
            
            if (player.x < padding) player.x = padding;
            
            if (currentScene === "luar") {
                if (!gateOpened && player.x >= gateWallX) { isTouchingGate = true; } else { isTouchingGate = false; }
                if (!gateOpened && player.x > gateWallX) player.x = gateWallX;
                
                if (player.x >= 1050 && player.x <= 1250) { isTouchingHouseDoor = true; } else { isTouchingHouseDoor = false; }
                if (player.x >= 1650 && player.x <= 1800) { isTouchingClawMachineLuar = true; } else { isTouchingClawMachineLuar = false; }

                if (!missionCompleted && player.x > houseWallRightX) player.x = houseWallRightX;
                
                let maxRightLimit = WORLD_WIDTH - padding;
                if (!keyClawActive) { maxRightLimit = 1800; }
                if (player.x > maxRightLimit) player.x = maxRightLimit;

                cameraX = player.x - canvas.width / 2 / zoomLevel;
                let maxCameraX = (WORLD_WIDTH - (canvas.width / zoomLevel));
                
                if (!missionCompleted) {
                    let lockedCameraX = 1200 - (canvas.width / zoomLevel) + 200; 
                    if (maxCameraX > lockedCameraX) maxCameraX = Math.max(0, lockedCameraX);
                }
                
                if (!keyClawActive) {
                    let lockedClawCamX = 1800 - (canvas.width / zoomLevel) + 200;
                    if (maxCameraX > lockedClawCamX) maxCameraX = Math.max(0, lockedClawCamX);
                }

                if (cameraX < 0) cameraX = 0;
                if (cameraX > maxCameraX) cameraX = maxCameraX;

            } else if (currentScene === "dalam_rumah") {
                cameraX = 0; 
                isTouchingGate = false;
                isTouchingHouseDoor = false;

                if (isPuzzleFinishedGlobal && isPhotoFinishedGlobal) {
                    document.getElementById('exitBtn').style.display = "block";
                } else {
                    document.getElementById('exitBtn').style.display = "none";
                }

                const innerHouseWidthBound = canvas.width / zoomLevel;
                if (player.x > innerHouseWidthBound - padding) {
                    player.x = innerHouseWidthBound - padding;
                }

                if (player.x >= 130 && player.x <= 600) { isTouchingTV = true; } else { isTouchingTV = false; }
                if (player.x >= 660 && player.x <= 790) { isTouchingRak = true; } else { isTouchingRak = false; }
                if (player.x >= 1050 && player.x <= 1200) { isTouchingRakFoto = true; } else { isTouchingRakFoto = false; }
            }
            
            const floatEffect = Math.sin(Date.now() / 300) * 5; 
            if (currentScene === "luar" && player.x >= 1880 && player.x < 1920) { player.yOffset = -30 + floatEffect; }
            else if (currentScene === "luar" && player.x >= 1920 && player.x < 2160) { player.yOffset = -55 + floatEffect; }
            else if (currentScene === "luar" && player.x >= 2160 && player.x < 2220) { player.yOffset = -30 + floatEffect; }
            else { player.yOffset = floatEffect; }

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            let pX = 0; let pY = 0;
            if (currentScene === "luar") {
                const drawWidth = WORLD_WIDTH * zoomLevel;
                const drawHeight = WORLD_HEIGHT * zoomLevel;
                const yOffset = (canvas.height - drawHeight) / 2;

                ctx.drawImage(imgAwan, -cameraX * 0.05 * zoomLevel, yOffset + awanYOffset, drawWidth, drawHeight * 0.8);
                ctx.drawImage(imgGunung, -cameraX * 0.4 * zoomLevel, yOffset + gunungYOffset, drawWidth, drawHeight);
                ctx.drawImage(imgTanah, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);
                ctx.drawImage(imgCaraBermain, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);
                ctx.drawImage(imgRumahMisi, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);

                if (imgMesinCapitLuar.complete) {
                    ctx.drawImage(imgMesinCapitLuar, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);
                }

                if (gateOpened) {
                    ctx.drawImage(imgSunflower, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);
                } else {
                    ctx.drawImage(imgGateTutup, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);
                    ctx.drawImage(imgSunflower, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);
                    if (isTouchingGate) { drawHelperTooltip("open the gate (Ent)", (480 - cameraX) * zoomLevel, yOffset + (drawHeight * 0.53)); }
                }

                if (isTouchingHouseDoor) { drawHelperTooltip("enter room (Ent)", ((houseDoorX + 110) - cameraX) * zoomLevel, yOffset + (drawHeight * 0.42)); }
                
                if (isTouchingClawMachineLuar) {

                        drawHelperTooltip("mainkan mesin capit (Ent)", (clawMachineX - cameraX) * zoomLevel, yOffset + (drawHeight * 0.45));

                        drawHelperTooltip("Gunakan Koin Anda di Mesin Capit 🪙", (clawMachineX - cameraX) * zoomLevel, yOffset + (drawHeight * 0.45));
                    }
                }

                pX = (player.x - cameraX) * zoomLevel - (player.width * zoomLevel / 2);
                pY = ((canvas.height * 0.8) - player.height) + (player.yOffset * zoomLevel) + (player.y * zoomLevel);
                const currentImg = (player.dir === 'right') ? kucingKanan : kucingKiri;
                ctx.drawImage(currentImg, pX, pY, player.width * zoomLevel, player.height * zoomLevel);

                if (gateOpened) { ctx.drawImage(imgGateBuka, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight); }
                ctx.drawImage(imgJembatan, -cameraX * zoomLevel, yOffset, drawWidth, drawHeight);

                ctx.drawImage(imgIsiRumahMisi, 0, 0, canvas.width, canvas.height);
                ctx.drawImage(imgTV, 0, 0, canvas.width, canvas.height);

                ctx.drawImage(imgSpeaker, 0, vibration, canvas.width, canvas.height);

                    photoFrameCounter++;
                    if (photoFrameCounter > 60) { photoFrameCounter = 0; tvPhotoIndex = (tvPhotoIndex + 1) % 10; }

                    if (tvPhotos[tvPhotoIndex] && tvPhotos[tvPhotoIndex].complete && tvPhotos[tvPhotoIndex].naturalWidth !== 0) {

                    } else {

                        ctx.fillStyle = "#ffffff"; ctx.font = "bold 24px 'Courier New'"; ctx.textAlign = "center"; ctx.textBaseline = "middle";
                        ctx.fillText("FOTO " + (tvPhotoIndex + 1), screenX + (screenW/2), screenY + (screenH/2));
                    }
                }

                if (isTouchingTV) {
                    const tooltipBaseX = canvas.width * 0.295; const tooltipBaseY = canvas.height * 0.25;
                    drawHelperTooltip(tvOn ? "(O) Matikan TV" : "(O) Nyalakan TV", tooltipBaseX, tooltipBaseY);
                    drawHelperTooltip("(P) Pilih Lagu", tooltipBaseX, tooltipBaseY + 30);
                }

                ctx.drawImage(imgRakPuzzle, 0, 0, canvas.width, canvas.height);
                if (isTouchingRak && !isPuzzleActive) {
                    drawHelperTooltip("(K) Buka Puzzle", canvas.width * 0.565, canvas.height * 0.50);
                }

                ctx.drawImage(imgRakFoto, 0, 0, canvas.width, canvas.height);
                if (isTouchingRakFoto && !isPhotoActive) {
                    drawHelperTooltip("(F) Urutkan Foto", canvas.width * 0.900, canvas.height * 0.35);
                }

                pX = (player.x) * zoomLevel - (player.width * zoomLevel / 2);
                pY = (canvas.height * 0.9) - (player.height * zoomLevel) + (player.y * zoomLevel);
                const currentImg = (player.dir === 'right') ? kucingKanan : kucingKiri;
                ctx.drawImage(currentImg, pX, pY, player.width * zoomLevel, player.height * zoomLevel);
            }

            // RENDERING ANIMASI LOVE GLOBAL
            if (loveAnimation.active) {
                ctx.save(); ctx.font = "bold 26px 'Courier New', monospace"; ctx.textAlign = "center";
                const hxBase = pX + (player.width * zoomLevel / 2) + 25; const hyBase = pY + 20; 
                loveAnimation.frameCounter++;
                if (loveAnimation.hearts.length < 3 && loveAnimation.frameCounter === (loveAnimation.hearts.length * loveAnimation.heartAppearDelay) + 1) {
                    let xShift = 0; 
                    if (loveAnimation.hearts.length === 1) xShift = -35; 
                    if (loveAnimation.hearts.length === 2) xShift = 35;  
                    loveAnimation.hearts.push({ xOffset: xShift, yOffset: 0, alpha: 1.0, speed: 0.6, colorPulse: 0 });
                }
                let allHeartsGone = true;
                for (let i = 0; i < loveAnimation.hearts.length; i++) {
                    const h = loveAnimation.hearts[i];
                    if (h.alpha > 0) {
                        allHeartsGone = false; h.yOffset -= h.speed; h.alpha -= 0.015; h.colorPulse += 0.1;
                        ctx.fillStyle = `rgba(${235 + Math.sin(h.colorPulse) * 20}, ${40 + Math.sin(h.colorPulse) * 40}, ${80 + Math.sin(h.colorPulse) * 40}, ${h.alpha})`;
                        ctx.shadowColor = "#ffffff"; ctx.shadowBlur = 2;
                        ctx.fillText("♥", hxBase + h.xOffset, hyBase + h.yOffset);
                    }
                }
                if (allHeartsGone && loveAnimation.hearts.length === 3) {
                    loveAnimation.loopCount++;
                    if (loveAnimation.loopCount < loveAnimation.loopLimit) { loveAnimation.hearts = []; loveAnimation.frameCounter = 0; } 
                    else { loveAnimation.active = false; loveAnimation.hearts = []; }
                }
                ctx.restore();
            }

            if (isClawPopupActive) { updateAndDrawClawMachineEngine(); }
            requestAnimationFrame(loop);
        }

        // --- SUB ENGINE UTAMA: ENGINE SIMULASI MESIN CAPIT ---
        function updateGameClawLogic() {
            let currentClawCenterX = 1320 + clawXOffset;
            let currentClawTopY = 320 + cakarYAdj; 

            if (cwGameState === "idle") {
                if (keys.left || isLeftBtnClicked) {
                    clawXOffset -= 8;
                    if (clawXOffset < -450) clawXOffset = -450;
                } else if (keys.right || isRightBtnClicked) {
                    clawXOffset += 8;
                    if (clawXOffset > 450) clawXOffset = 450;
                }
                mainClawYOffset = 0;
                heartX = 1450;
                heartY = 1250;
                heartCaughtStatus = false;
            }

            if (cwGameState === "descending") {
                mainClawYOffset += 12;
                if (currentClawTopY + mainClawYOffset + (cakarHeight * 0.5) >= heartY) {
                    cwGameState = "checking";
                }
            }

            if (cwGameState === "checking") {
                let centerHeartX = heartX + (hatiWidth / 2);
                let matchDistance = Math.abs(currentClawCenterX - centerHeartX);

                if (matchDistance < 100) {
                    heartCaughtStatus = true;
                    heartX = currentClawCenterX - (hatiWidth / 2) - 18;
                    heartY = currentClawTopY + mainClawYOffset + (cakarHeight * 0.25);
                    cwGameState = "ascending";
                } else {
                    heartCaughtStatus = false;
                    cwGameState = "ascending";
                }
            }

            if (cwGameState === "ascending") {
                mainClawYOffset -= 12;
                if (heartCaughtStatus) {
                    heartX = currentClawCenterX - (hatiWidth / 2) - 18;
                    heartY = currentClawTopY + mainClawYOffset + (cakarHeight * 0.25);
                }
                if (mainClawYOffset <= 0) {
                    mainClawYOffset = 0;
                    cwGameState = heartCaughtStatus ? "rewinding" : "idle";
                }
            }

            if (cwGameState === "rewinding") {
                clawXOffset -= 10;
                if (heartCaughtStatus) {
                    heartX = currentClawCenterX - (hatiWidth / 2) - 18;
                    heartY = currentClawTopY + mainClawYOffset + (cakarHeight * 0.25);
                }
                if (clawXOffset <= -450) {
                    clawXOffset = -450;
                    cwGameState = "dropping";
                }
            }

            if (cwGameState === "dropping") {
                heartX = currentClawCenterX - (hatiWidth / 2) - 18;
                heartY += 18;
                if (heartY >= 1350) {
                    heartY = 1350;
                    isClawPopupActive = false;
                    document.getElementById('clawMachinePopup').style.display = 'none';
                    
                    if (!keyClawActive) {
                        triggerGlobalItemReward(3, "MISI CAPIT SELESAI! ✨🔑");
                    } else {
                        triggerGlobalItemReward(4, "KAMU BERHASIL LAGI! ✨💖");
                    }
                    startLoveAnimation(); 
                }
            }
        }

        function updateAndDrawClawMachineEngine() {
            updateGameClawLogic();

            if (imgMachine.complete && imgMain.complete && imgBar.complete && 
                imgTumpukanHati.complete && imgLeftBtn.complete && imgRightBtn.complete &&

                if (keys.left || isLeftBtnClicked) clawCtx.translate(0, 15);
                clawCtx.drawImage(imgLeftBtn, 0, 0, size, size);
                clawCtx.restore();

                clawCtx.save();
                if (keys.right || isRightBtnClicked) clawCtx.translate(0, 15);
                clawCtx.drawImage(imgRightBtn, 0, 0, size, size);
                clawCtx.restore();

                clawCtx.save();
                if (cwGameState === "descending" || isMainBtnClicked) clawCtx.translate(0, 15);
                clawCtx.drawImage(imgMainBtn, 0, 0, size, size);
                clawCtx.restore();

                clawCtx.drawImage(imgBar, 0, 0, size, size);
                clawCtx.drawImage(imgTumpukanHati, 0, 0, size, size);

                clawCtx.save();
                if (heartCaughtStatus) {
                    clawCtx.shadowColor = "#ff619b";
                    clawCtx.shadowBlur = 80;
                }
                clawCtx.drawImage(imgHati, heartX, heartY, hatiWidth, hatiHeight);
                clawCtx.restore();

                clawCtx.save();
                clawCtx.translate(clawXOffset, 0);
                clawCtx.drawImage(imgClaw, 0, 0, size, size);
                clawCtx.restore();

                clawCtx.save();
                clawCtx.translate(1320 + clawXOffset - (cakarWidth / 2), 320 + mainClawYOffset + cakarYAdj);
                clawCtx.drawImage(imgMainClaw, 0, 0, cakarWidth, cakarHeight);
                clawCtx.restore();

                clawCtx.drawImage(imgOutBox, 0, 0, size, size);
            }
        }

            if (cwGameState !== "idle") return;
            const rect = clawCanvas.getBoundingClientRect();
            const scaleX = size / rect.width;
            const scaleY = size / rect.height;
            const clickX = (e.clientX - rect.left) * scaleX;
            const clickY = (e.clientY - rect.top) * scaleY;

            if (clickX >= 700 && clickX <= 960 && clickY >= 2100 && clickY <= 2450) isLeftBtnClicked = true;
            if (clickX >= 1050 && clickX <= 1300 && clickY >= 2100 && clickY <= 2450) isRightBtnClicked = true;
            if (clickX >= 1550 && clickX <= 1950 && clickY >= 2100 && clickY <= 2450) {
                isMainBtnClicked = true;
                cwGameState = "descending";
            }
        });

            isLeftBtnClicked = false;

        // --- HANDLER KEYBOARD INPUT GAME UTAMA & MESIN CAPIT ---
        window.addEventListener('keydown', e => {
            if (e.key === 'ArrowLeft' || e.key === 'a') keys.left = true;
            if (e.key === 'ArrowRight' || e.key === 'd') keys.right = true;
            if (e.key === 'ArrowUp' || e.key === 'w' || e.key === ' ') keys.up = true;
            
            if (e.key === 'Enter') {
                if (isPopupActive) {
                    checkPin();
                } else if (isClawPopupActive && cwGameState === "idle") {
                    cwGameState = "descending";
                } else if (currentScene === "luar" && isTouchingClawMachineLuar) {
                    if (keyClawActive) {
                        isClawPopupActive = true;
                        cwGameState = "idle";
                        document.getElementById('clawMachinePopup').style.display = 'flex';
                    }
                } else if (currentScene === "luar" && isTouchingHouseDoor) {
                    currentScene = "dalam_rumah";
                    player.x = 230; player.y = 0;
                } else if (currentScene === "luar" && isTouchingGate) {
                    openPinPopup();
                }
            }

            if (currentScene === "dalam_rumah") {
                if (isTouchingTV) {
                    if (e.key.toLowerCase() === 'o') {
                        tvOn = !tvOn;
                        if (tvOn) { playSong(1); } else { document.getElementById('bgmAudioActual').pause(); updateCatalogUI(); }
                    }
                    if (e.key.toLowerCase() === 'p') {
                        const panel = document.getElementById('songPanel');
                        panel.style.display = (panel.style.display === 'block') ? 'none' : 'block';
                        updateCatalogUI();
                    }
                }

                if (isTouchingRak && e.key.toLowerCase() === 'k') {
                    isPuzzleActive = !isPuzzleActive;
                    if (isPuzzleActive) {
                        closePhotoGame();
                        document.getElementById('puzzlePopup').style.display = 'flex';
                        initPuzzleGame(); 
                    } else { closePuzzleGame(); }
                }

                if (isTouchingRakFoto && e.key.toLowerCase() === 'f') {
                    isPhotoActive = !isPhotoActive;
                    if (isPhotoActive) {
                        closePuzzleGame();
                        document.getElementById('photoPopup').style.display = 'flex';
                        initPhotoGame();
                    } else { closePhotoGame(); }
                }
            }
            
            if (e.key === 'Escape' && currentScene === "dalam_rumah" && isPuzzleFinishedGlobal && isPhotoFinishedGlobal) { exitHouse(); }
        });

        window.addEventListener('keyup', e => {
            if (e.key === 'ArrowLeft' || e.key === 'a') keys.left = false;
            if (e.key === 'ArrowRight' || e.key === 'd') keys.right = false;
            if (e.key === 'ArrowUp' || e.key === 'w' || e.key === ' ') keys.up = false;

                document.onmousemove = null;

                    const slotEl = document.getElementById("puzzleTargetSlot_" + i);

                    if (currentMouseX >= sb.left && currentMouseX <= sb.right &&

                            oldImg.style.transform = 'scale(0.395)';

                        elmnt.style.transform = 'scale(1)';
                        elmnt.style.left = '0px';
                        elmnt.style.top = '0px';

                    }

                    poolEl.appendChild(elmnt);

                    elmnt.style.transform = 'scale(0.395)';

                    if (dropY < 0) dropY = 0; if (dropY > 210 - 105) dropY = 210 - 105;
                    elmnt.style.left = dropX + "px";
                    elmnt.style.top = dropY + "px";
                    elmnt.style.zIndex = "10";
                }
            }
        }

        function checkPuzzleSolution() {
            let totalPlaced = 0;
            let isCorrect = true;
            for (let i = 1; i <= 16; i++) {
                const slotEl = document.getElementById("puzzleTargetSlot_" + i);
                if (slotEl && slotEl.children.length > 0) {
                    totalPlaced++;
                    let pieceId = parseInt(slotEl.children[0].getAttribute('data-id'));
                    if (pieceId !== i) isCorrect = false;
                }
            }
            if (totalPlaced < 16) {
                showCustomAlert("Letakkan semua 16 kepingan puzzle ke semua kotak target terlebih dahulu!");
                return;
            }
            if (isCorrect) {
                isPuzzleFinishedGlobal = true; 
                closePuzzleGame();
                triggerGlobalItemReward(1, "PUZZLE SELESAI! ✨💰");
                startLoveAnimation(); 
            } else {
                showCustomAlert("Hmm, susunan gambarnya masih ada yang keliru. Cek lagi ya!");
            }
        }

        // --- GLOBAL ITEM REWARD SYSTEM MANAGER ---
        function triggerGlobalItemReward(itemNumber, titleText) {
            document.getElementById('successMainTitle').innerText = titleText;
            const textPopup = document.getElementById('successTextPopup');
            textPopup.style.display = 'block';

            let targetEl;
            if (itemNumber === 1) { targetEl = document.getElementById('coinReward1'); coin1Active = true; }
            else if (itemNumber === 2) { targetEl = document.getElementById('coinReward2'); coin2Active = true; }
            else if (itemNumber === 3) { targetEl = document.getElementById('keyRewardClaw'); keyClawActive = true; }
            else if (itemNumber === 4) { targetEl = document.getElementById('rafizaRewardClaw'); }

            targetEl.style.display = 'block';
            targetEl.style.top = '50%';
            targetEl.style.left = '50%';

            // LOGIKA PEMILIHAN ANIMASI: RAFIZA SENANG AKAN DI-SCALE LEBIH KECIL (0.7)
            if (itemNumber === 4) {
                targetEl.style.animation = 'rafizaZoomScreen 1.5s ease-in-out forwards';
            } else {
                targetEl.style.animation = 'coinZoomScreenInAndOut 1.5s ease-in-out forwards';
            }

            setTimeout(() => {
                targetEl.style.animation = ''; 
                textPopup.style.display = 'none'; 
                
                if (itemNumber !== 4) {
                    rearrangeItemsLayout(); 
                    targetEl.style.animation = 'coinBounceTwice 0.6s ease-out 2';
                    setTimeout(() => {
                        targetEl.style.animation = '';
                        makeElementDraggable(targetEl, itemNumber); 
                    }, 1200);
                } else {
                    targetEl.style.display = 'none';
                    resetClawMachineGameplayState();
                }
            }, 1500);
        }

        function rearrangeItemsLayout() {
            let activeElements = [];
            if (coin1Active) activeElements.push(document.getElementById('coinReward1'));
            if (coin2Active) activeElements.push(document.getElementById('coinReward2'));
            if (keyClawActive) activeElements.push(document.getElementById('keyRewardClaw'));

            activeElements.forEach((itemEl, index) => {
                itemEl.style.top = 'calc(100vh - 170px)';
                itemEl.style.left = (-45 + (index * 80)) + 'px';
                itemEl.style.transform = 'scale(0.3)';
            });
        }

        function makeElementDraggable(elmnt, itemNumber) {
            let pos1 = 0, pos2 = 0, pos3 = 0, pos4 = 0;
            elmnt.onmousedown = dragMouseDown;

            function dragMouseDown(e) {
                e = e || window.event;
                e.preventDefault();
                pos3 = e.clientX;
                pos4 = e.clientY;
                document.onmouseup = closeDragElement;
                document.onmousemove = elementDrag;
            }

            function elementDrag(e) {
                e = e || window.event;
                e.preventDefault();
                pos1 = pos3 - e.clientX;
                pos2 = pos4 - e.clientY;
                pos3 = e.clientX;
                pos4 = e.clientY;
                elmnt.style.top = (elmnt.offsetTop - pos2) + "px";
                elmnt.style.left = (elmnt.offsetLeft - pos1) + "px";
            }

            function closeDragElement() {
                document.onmouseup = null;
                document.onmousemove = null;
                let currentMouseX = pos3;
                let currentMouseY = pos4;

                if (currentScene === "luar" && !keyClawActive && (itemNumber === 1 || itemNumber === 2)) {
                    let screenClawX = (clawMachineX - cameraX) * zoomLevel;
                    let screenClawY = (canvas.height * 0.8) - 50; 

                    if (currentMouseX >= screenClawX - 150 && currentMouseX <= screenClawX + 150 &&
                        currentMouseY >= screenClawY - 250 && currentMouseY <= screenClawY + 250) {
                        
                        elmnt.style.display = 'none';
                        if (itemNumber === 1) coin1Active = false; 
                        else if (itemNumber === 2) coin2Active = false;
                        
                        isClawPopupActive = true;
                        cwGameState = "idle";
                        document.getElementById('clawMachinePopup').style.display = 'flex';
                    }
                }
                rearrangeItemsLayout(); 
            }
        }
